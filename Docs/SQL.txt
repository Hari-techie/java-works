CREATE TABLE CITY
(ID NUMBER,
COUNTRY_ID NUMBER,
CITY_NAME VARCHAR(50),
LAST_UPDATE TIMESTAMP DEFAULT CURRENT_TIMESTAMP);

ALTER TABLE CITY ADD CONSTRAINT PK_ID PRIMARY KEY(ID);

DESC CITY;
SELECT * FROM CITY;
